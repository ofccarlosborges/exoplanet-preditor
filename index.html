# backend_server.py from flask import Flask, request, jsonify, render_template_string from flask_cors import CORS import pandas as pd import numpy as np import tensorflow as tf import joblib import io import base64 app = Flask(__name__) CORS(app) # Permite requisi√ß√µes do frontend # Carregar modelo e scaler treinados print("Carregando modelo...") model = tf.keras.models.load_model('modelo_massa_exoplaneta.h5') scaler = joblib.load('scaler_exoplanetas.pkl') print("‚úÖ Modelo carregado com sucesso!") # Colunas necess√°rias (mesmas do seu treinamento) REQUIRED_COLUMNS = [ 'pl_orbper', 'pl_orbsmax', 'pl_radj', 'pl_orbeccen', 'pl_eqt', 'pl_orblper', 'pl_rvamp', 'st_teff', 'st_rad', 'st_mass', 'st_met', 'st_logg' ] @app.route('/') def home(): # Retorna o HTML do frontend return render_template_string(HTML_TEMPLATE) @app.route('/predict', methods=['POST']) def predict(): try: # Receber dados do frontend data = request.json csv_data = data['csv_data'] # Converter para DataFrame df = pd.DataFrame(csv_data) # Verificar colunas necess√°rias missing_cols = [col for col in REQUIRED_COLUMNS if col not in df.columns] if missing_cols: return jsonify({ 'error': f'Colunas obrigat√≥rias n√£o encontradas: {missing_cols}' }), 400 # Filtrar e limpar dados df_clean = df[REQUIRED_COLUMNS].copy() df_clean = df_clean.dropna() if len(df_clean) == 0: return jsonify({'error': 'Nenhum dado v√°lido encontrado'}), 400 # Preparar dados para predi√ß√£o X = df_clean.values # Padronizar usando o scaler treinado X_scaled = scaler.transform(X) # Fazer predi√ß√µes com o modelo real predictions = model.predict(X_scaled, verbose=0) predictions = predictions.flatten() # Calcular estat√≠sticas stats = { 'total_planets': len(predictions), 'avg_mass': float(np.mean(predictions)), 'min_mass': float(np.min(predictions)), 'max_mass': float(np.max(predictions)), 'std_mass': float(np.std(predictions)) } # Classificar planetas categories = [] for mass in predictions: if mass < 0.1: categories.append('Rochoso') elif mass < 0.3: categories.append('Super-Terra') elif mass < 4: categories.append('Netuno') else: categories.append('Gigante Gasoso') # Preparar resultados detalhados detailed_results = [] for i, (_, row) in enumerate(df_clean.iterrows()): detailed_results.append({ 'index': i + 1, 'orbital_period': float(row['pl_orbper']), 'radius': float(row['pl_radj']), 'stellar_mass': float(row['st_mass']), 'predicted_mass': float(predictions[i]), 'category': categories[i], 'confidence': 'Alta' if predictions[i] > 0.1 else 'M√©dia' }) return jsonify({ 'success': True, 'stats': stats, 'predictions': predictions.tolist(), 'categories': categories, 'detailed_results': detailed_results }) except Exception as e: return jsonify({'error': str(e)}), 500 @app.route('/model_info', methods=['GET']) def model_info(): """Retorna informa√ß√µes sobre o modelo""" try: # Informa√ß√µes do modelo model_info = { 'layers': len(model.layers), 'parameters': model.count_params(), 'input_shape': model.input_shape, 'output_shape': model.output_shape, 'optimizer': model.optimizer.get_config()['name'], 'loss': model.loss, 'required_columns': REQUIRED_COLUMNS } return jsonify(model_info) except Exception as e: return jsonify({'error': str(e)}), 500 #
